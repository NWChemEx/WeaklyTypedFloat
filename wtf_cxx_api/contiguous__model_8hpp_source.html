244template<typename TupleType, typename Visitor, typename... Args>
245auto visit_contiguous_model(Visitor&& visitor, Args&&... args) {
246 auto lambda = [&](auto&&... inner_args) {
247 return visitor(inner_args.span()...);
248 };
249 return wtf::detail_::dispatch<ContiguousModel, TupleType>(
250 lambda, std::forward<Args>(args)...);
