43template<std::size_t I, typename VariantType, typename T>
44VariantType downcast_impl(T&& holder) {
45 // Is holder a read-only object (regardless of whether it is holding a
46 // read-only type)?
47 using clean_type = std::remove_reference_t<T>;
48 static constexpr bool is_const = std::is_const_v<clean_type>;
49
50 if constexpr(I == std::variant_size_v<VariantType>) {
51 throw std::runtime_error("Type not in variant");
52 } else {
53 // This iteration we try to cast to TargetType
54 using TargetType = std::variant_alternative_t<I, VariantType>;
55
56 // Is TargetType a read-only type?
57 using clean_target_type = std::remove_pointer_t<TargetType>;
58 static constexpr bool const_target = std::is_const_v<clean_target_type>;
59
60 if constexpr(is_const && !const_target) {
61 // Can't cast const to non-const
62 return downcast_impl<I + 1, VariantType>(std::forward<T>(holder));
63 } else {
64 if(auto* p = dynamic_cast<TargetType>(&holder)) {
65 return VariantType{p};
66 } else {
67 return downcast_impl<I + 1, VariantType>(
68 std::forward<T>(holder));
69 }
70 }
71 }
72}

93template<template<concepts::FloatingPoint> typename ModelType, typename Types,
94 typename T>
95auto downcaster(T&& holder) {
96 constexpr bool is_const = std::is_const_v<std::remove_reference_t<T>>;
97 using variant_t = typename VariantFromTuple<ModelType, Types>::value;
98 using cvariant_t = typename ConstVariantFromTuple<ModelType, Types>::value;
99 using variant_type = std::conditional_t<is_const, cvariant_t, variant_t>;
100 return downcast_impl<0, variant_type>(std::forward<T>(holder));
101}

122 using variant_t = typename VariantFromTuple<ModelType, Types>::cvalue;
123 using cvariant_t = typename ConstVariantFromTuple<ModelType, Types>::cvalue;
124 using variant_type = std::conditional_t<is_const, cvariant_t, variant_t>;
