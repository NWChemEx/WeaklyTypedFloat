399template<typename TupleType>
400auto make_float_buffer(std::vector<FloatBuffer::value_type> buffer) {
401 FloatBuffer rv;
402 if(buffer.size() == 0) return rv;
403 auto initializer = [&](auto value0) {
404 using T = std::decay_t<decltype(value0)>;
405 std::vector<T> temp_buffer(buffer.size());
406 rv = FloatBuffer(std::move(temp_buffer));
407 rv.at(0) = value0;
408 };
409 fp::visit_float<TupleType>(initializer, buffer[0]);
410 for(std::size_t i = 1; i < buffer.size(); ++i) {
411 auto appender = [&](auto value) { rv.at(i) = value; };
412 fp::visit_float<TupleType>(appender, buffer[i]);
413 }
414 return rv;
415}
