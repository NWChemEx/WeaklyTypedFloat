204template<typename TypeTuple>
205bool TypeHolder::is_implicitly_convertible_to(const TypeHolder& other) const {
206 auto lambda = [](auto&& from, auto&& to) {
207 using from_type = typename std::decay_t<decltype(from)>::value_type;
208 using to_type = typename std::decay_t<decltype(to)>::value_type;
209 using clean_from_type = std::decay_t<from_type>;
210 using clean_to_type = std::decay_t<to_type>;
211 return wtf::type_traits::is_convertible_v<clean_from_type,
212 clean_to_type>;
213 };
214 // Cannot convert away const-qualification
215 if(is_const() && !other.is_const()) return false;
216 return wtf::detail_::dispatch<TypeModel, TypeTuple>(lambda, *this, other);
217}
