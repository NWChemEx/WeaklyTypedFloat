201template<typename TypeTuple>
202bool TypeHolder::is_implicitly_convertible_to(const TypeHolder& other) const {
203 // Cannot convert away const-qualification
204 if(is_const() && !other.is_const()) return false;
205
206 // Handle nullptr_t conversions
207 bool this_null = this->is_nullptr();
208 bool other_null = other.is_nullptr();
209 if(this_null || other_null) { return this_null && other_null; }
210
211 auto lambda = [](auto&& from, auto&& to) {
212 using from_type = typename std::decay_t<decltype(from)>::value_type;
213 using to_type = typename std::decay_t<decltype(to)>::value_type;
214 using clean_from_type = std::decay_t<from_type>;
215 using clean_to_type = std::decay_t<to_type>;
216 return wtf::type_traits::is_convertible_v<clean_from_type,
217 clean_to_type>;
218 };
219
220 return wtf::detail_::dispatch<TypeModel, TypeTuple>(lambda, *this, other);
221}
