115 template<typename FxnType>
116 auto run(FxnType&& fxn) {
117 return std::apply(
118 [&](auto&&... args) {
119 return run_impl_(std::forward<FxnType>(fxn),
120 std::forward<decltype(args)>(args)...);
121 },

123 }

144 template<typename FxnType, typename... Args>
145 auto run_impl_(FxnType&& fxn, Args&&... args) {
146 auto deref_pointers = [&](auto&&... args) {
147 return fxn(*std::forward<decltype(args)>(args)...);
148 };
149
150 return std::visit(deref_pointers, std::forward<Args>(args)...);
151 }

187template<template<concepts::FloatingPoint> typename ModelType,
188 typename TypeTuple, typename FxnType, typename VariantTuple,

190auto dispatch_impl_common(FxnType&& fxn, DispatchHelper<VariantTuple> helper,
191 HolderTypes&&... other_holders) {
192 constexpr auto n_other_holders = sizeof...(other_holders);
193 if constexpr(n_other_holders > 0) {
194 return dispatch_impl<ModelType, TypeTuple>(
195 std::forward<FxnType>(fxn), std::move(helper),
196 std::forward<HolderTypes>(other_holders)...);
197 } else {
198 return helper.run(std::forward<FxnType>(fxn));
199 }

235template<template<concepts::FloatingPoint> typename ModelType,
236 typename TypeTuple, typename FxnType, typename VariantTuple,
237 typename HolderType, typename... HolderTypes>
238auto dispatch_impl(FxnType&& fxn, DispatchHelper<VariantTuple> helper,
239 HolderType&& h, HolderTypes&&... other_holders) {
240 const bool is_const = h.is_const();
241
242 // Because we must call the correct downcaster based on cv-qualifiers that
243 // are only knowable at runtime, and because those downcasters all have
244 // different return types (and thus signatures) we must explicitly branch.
245
246 if(is_const) {
247 auto v =
248 const_downcaster<ModelType, TypeTuple>(std::forward<HolderType>(h));
249 auto new_helper = helper.add_variant(std::move(v));
250 return dispatch_impl_common<ModelType, TypeTuple>(
251 std::forward<FxnType>(fxn), std::move(new_helper),
252 std::forward<HolderTypes>(other_holders)...);
253 } else {
254 auto v = downcaster<ModelType, TypeTuple>(std::forward<HolderType>(h));
255 auto new_helper = helper.add_variant(std::move(v));
256 return dispatch_impl_common<ModelType, TypeTuple>(
257 std::forward<FxnType>(fxn), std::move(new_helper),
258 std::forward<HolderTypes>(other_holders)...);
259 }
260}

290template<template<concepts::FloatingPoint> typename ModelType,
291 typename TypeTuple, typename FxnType, typename... HolderTypes>
292auto dispatch(FxnType&& fxn, HolderTypes&&... other_holders) {
293 if constexpr(sizeof...(other_holders) == 0) {
294 // If there are no holders, just call the callable
295 return fxn();
296 } else {
297 using empty_variant_tuple = std::tuple<>;
298 DispatchHelper<empty_variant_tuple> helper(empty_variant_tuple{});
299 return dispatch_impl<ModelType, TypeTuple>(
300 std::forward<FxnType>(fxn), std::move(helper),
301 std::forward<HolderTypes>(other_holders)...);
302 }
